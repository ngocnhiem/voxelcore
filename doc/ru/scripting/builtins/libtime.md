# Библиотека time

```lua
-- Возвращает время с момента запуска движка в секундах и милисекундах.
time.uptime() -> number

-- Возвращает дельту времени в секундах и милисекундах (время прошедшее с предыдущего кадра)
time.delta() -> number

-- Возвращает время UTC в секундах
time.utc_time() -> int

-- Возвращает локальное (системное) время в секундах
time.local_time() -> int

-- Возвращает смещение локального времени от UTC в секундах
time.utc_offset() -> int
```
